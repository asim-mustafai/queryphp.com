# 查询多条数据.findAll

Testing Is Documentation

tests/Database/Read/FindAllTest.php

Uses

  • use Tests\Database\DatabaseTestCase as TestCase;

# findAll 查询多条数据

public function testBaseUse(): void
{
    $connect = $this->createDatabaseConnectMock();
    $sql = <<<'eot'
        [
            "SELECT `test`.* FROM `test`",
            [],
            false,
            null,
            null,
            []
        ]
        eot;

    $this->assertSame(
        $sql,
        $this->varJson(
            $connect
                ->sql()
                ->table('test')
                ->findAll()
        )
    );
}
